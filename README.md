## Persiapan & Instalasi

1.  **Clone Repository**
    ```bash
    git clone <repository-url>
    cd landingpage
    ```

2.  **Install Dependencies**
    ```bash
    npm install
    ```

3.  **Konfigurasi Environment Variables**
    Buat file `.env` di root project dan tambahkan konfigurasi Supabase Anda:
    ```env
    VITE_SUPABASE_URL=your_supabase_project_url
    VITE_SUPABASE_ANON_KEY=your_supabase_anon_key
    ```

4.  **Jalankan Aplikasi**
    ```bash
    npm run dev
    ```

## Setup Database (Supabase)

Jalankan script SQL berikut di SQL Editor Supabase Anda untuk membuat tabel dan policy yang diperlukan.

```sql
-- 1. Tabel Agendas
create table agendas (
  id bigint generated by default as identity primary key,
  title text not null,
  slug text unique not null,
  content text,
  image_url text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 2. Tabel Documentation Items
create table documentation_items (
  id bigint generated by default as identity primary key,
  title text not null,
  type text check (type in ('image', 'video')) not null,
  url text not null,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 3. Tabel Page Views (Analytics)
create table page_views (
  id bigint generated by default as identity primary key,
  page_path text not null,
  user_agent text,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null
);

-- 4. Storage Buckets
insert into storage.buckets (id, name, public) values ('agenda-images', 'agenda-images', true);
insert into storage.buckets (id, name, public) values ('documentation', 'documentation', true);

-- 5. Row Level Security (RLS) Policies
alter table agendas enable row level security;
alter table documentation_items enable row level security;
alter table page_views enable row level security;

-- Policies for Agendas
create policy "Public view agendas" on agendas for select to public using (true);
create policy "Admin manage agendas" on agendas for all to authenticated using (true);

-- Policies for Documentation
create policy "Public view documentation" on documentation_items for select to public using (true);
create policy "Admin manage documentation" on documentation_items for all to authenticated using (true);

-- Policies for Page Views
create policy "Public track views" on page_views for insert to public with check (true);
create policy "Admin view stats" on page_views for select to authenticated using (true);

-- Policies for Storage
create policy "Public view storage" on storage.objects for select to public using (bucket_id in ('agenda-images', 'documentation'));
create policy "Admin manage storage" on storage.objects for all to authenticated using (bucket_id in ('agenda-images', 'documentation'));
```
